```bash
npm install --save fef-utils
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'fef-utils'
import 'fef-utils/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
